{
    int x1, x2, x3, x4, x5;

    cout << "Enter 5 Numbers : ";

Q6...
swap the values of a and b

#include <bits/stdc++.h>
using namespace std;

int main()
{
    int a, b;

    cout << "Enter two numbers A and B respectively : ";
    cin >> a >> b;

    int c;
    c = a;
    a = b;
    b = c;

    cout << "After swap, A = " << a << " and B = " << b;

    return 0;
}

Q7...
take an input from user and check whether it is positive negative or 0.

#include <bits/stdc++.h>
using namespace std;

int main()
{
    int x;
    cout << "Enter a number : ";
    cin >> x;

    if (x > 0)
    {
        cout << x << " is Positive";
    }
    else if (x < 0)
    {
        cout << x << " is Negative";
    }
    else
    {
        cout << x << " -  zero";
    }

    return 0;
}

    }
    else
    {

    }
